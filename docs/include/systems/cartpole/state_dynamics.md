$$
\diff \left(
    \begin{array}{c}
       \vartheta \\
        x \\
        \omega  \\
        v_x 
    \end{array}
    \right) = \left(
    \begin{array}{c}
        \omega \\
        v_x \\
        \varepsilon  \\
        \frac{F + m_p l (\omega ^2 \sin{\vartheta} - \varepsilon  \cos{\vartheta})}{m_c + m_p}
    \end{array}
    \right) \diff t
$$

where we denoted

$$
\varepsilon = g \sin{\vartheta} - \cos{\vartheta}\frac{F + m_p l \omega^2 \sin{\vartheta}}{\frac{4l}{3}(m_c + m_p) - lm_p \cos^2{\vartheta}}
$$
